cmake_minimum_required(VERSION 2.8)
project(hello_py)
# Find necessary packages
find_package(PythonLibs 3.5 REQUIRED)
set(PYTHON_INCLUDE_DIR "/usr/include/python3.5")
include_directories(${PYTHON_INCLUDE_DIR})
find_package(Boost COMPONENTS python3 REQUIRED)
include_directories(${Boost_INCLUDE_DIR})
set(CMAKE_CXX_STANDARD 11)
set(SOURCE_FILES hello_py.cc hello.cpp)
add_library(hello_py SHARED ${SOURCE_FILES})
target_link_libraries(hello_py ${Boost_LIBRARIES} ${PTYHON_LIBRARIES}) 
#add_dependencies(greet_ext Boost)
set_target_properties(hello_py PROPERTIES SUFFIX .so)
set_target_properties(hello_py PROPERTIES PREFIX "")
